void CAN_filter_config_init(void) {
    CAN_FilterTypeDef canfilterconfig;

    canfilterconfig.FilterActivation = CAN_FILTER_ENABLE;
    canfilterconfig.FilterBank = 0;
    canfilterconfig.FilterFIFOAssignment = CAN_FILTER_FIFO0;
    canfilterconfig.FilterIdHigh = 0x446 << 5;
    canfilterconfig.FilterIdLow = 0x0000;
    canfilterconfig.FilterMaskIdHigh = 0x7FF << 5; // Proper full mask for standard ID
    canfilterconfig.FilterMaskIdLow = 0x0000;
    canfilterconfig.FilterMode = CAN_FILTERMODE_IDMASK;
    canfilterconfig.FilterScale = CAN_FILTERSCALE_32BIT;

    if (HAL_CAN_ConfigFilter(&hcan, &canfilterconfig) != HAL_OK) {
        Error_Handler();
    }
}

void HAL_CAN_RxFifo0MsgPendingCallback(CAN_HandleTypeDef *hcan) {
    if (HAL_CAN_GetRxMessage(hcan, CAN_RX_FIFO0, &rx_header, rx_buffer) != HAL_OK) {
        Error_Handler();
    }

    if (rx_header.StdId == 0x446) {
        rx_flag = 1;
        HAL_UART_Transmit(&huart1, rx_buffer, sizeof(rx_buffer), 1000);
    }
}

void CAN_transmit_init(void) {
    tx_header.IDE = CAN_ID_STD;
    tx_header.StdId = 0x446;
    tx_header.RTR = CAN_RTR_DATA;
    tx_header.DLC = 8;
}

void HAL_UART_RxCpltCallback(UART_HandleTypeDef *huart) {
    if (huart->Instance == USART1) {
        uart_rx_flag = 1;
        rx_flag = 0;
        HAL_UART_Receive_IT(&huart1, rx_buffer, sizeof(rx_buffer));
    }
}

void transmit_data_block(void) {
    if (uart_rx_flag == 1) {
        memcpy(tx_buffer, rx_buffer, sizeof(rx_buffer));

        HAL_StatusTypeDef status = HAL_CAN_AddTxMessage(&hcan, &tx_header, tx_buffer, &TxMailbox);

        if (status == HAL_OK) {
            strcpy(response, "OK\n");
        } else {
            strcpy(response, "FAIL\n");
        }

        HAL_UART_Transmit_IT(&huart1, (uint8_t *)response, strlen(response));
        uart_rx_flag = 0;
    }
}

void HAL_UART_TxCpltCallback(UART_HandleTypeDef *huart) {
    if (huart->Instance == USART1) {
        tx_flag = 1;
    }
}
