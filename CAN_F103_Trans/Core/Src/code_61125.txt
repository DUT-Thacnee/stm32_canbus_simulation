volatile uint8_t blink_pb11 = 0;

uint8_t pb10_state = 0;

uint32_t last_toggle_pb10 = 0;

volatile uint8_t can_send_adc_high = 0;

int8_t  tx_buffer[8] = {0};
volatile GPIO_PinState hall_last_state = GPIO_PIN_RESET;
void transmit_data_block(void);

/* Cấu hình header CAN để truyền */
void CAN_transmit_init(void) {
	tx_header.DLC = 8;
    tx_header.IDE = CAN_ID_STD;
    tx_header.RTR = CAN_RTR_DATA;
    tx_header.StdId = 0x111;

}

void HAL_GPIO_EXTI_Callback(uint16_t GPIO_Pin)
{
    current_time = HAL_GetTick();
    if (current_time - last_interrupt_time < DEBOUNCE_DELAY)
        return;
    last_interrupt_time = current_time;

    switch(GPIO_Pin)
    {
    case GPIO_PIN_6:
               hall = HAL_GPIO_ReadPin(GPIOB, GPIO_PIN_6);
               if (hall != hall_last_state) {
                   hall_last_state = hall;
                   tx_buffer[2] = hall;
                   transmit_data_block();
               }
               break;
        case GPIO_PIN_10:  // PB10 → PB11
            tx_buffer[0] = ( tx_buffer[0] == 1) ? 0 : 1;
            blink_pb11 ^= 1;
            transmit_data_block();
            break;

        case GPIO_PIN_9:  // PB9 → PB10
        	 tx_buffer[0] = ( tx_buffer[0] == 2) ? 0 : 2;
            blink_pb10 ^= 1;
            transmit_data_block();
            break;

        case GPIO_PIN_15: //
        	 tx_buffer[0] = ( tx_buffer[0] == 3) ? 0 : 3;
        	 blink_pb10 ^= 1;
        	 blink_pb11 ^= 1;
        	 transmit_data_block();
    }
}

/* Truyền dữ liệu từ UART → CAN */
void transmit_data_block(void) {
        HAL_CAN_AddTxMessage(&hcan, &tx_header, (uint8_t *)tx_buffer, &TxMailbox);
}

void HAL_ADC_ConvCpltCallback(ADC_HandleTypeDef* hadc) {
    // Dùng VR[0] là đầu vào biến trở để xác định mức
    if (VR[0] < 50) {
        can_send_adc_low = 1;
        tx_buffer[1] = 0;
    } else if (VR[0] > 3000) {
        can_send_adc_high = 1;
        tx_buffer[1] = 1;
    }
    HAL_ADC_Start_DMA(&hadc1, VR, 1);
}

/**
  * @brief  The application entry point.
  * @retval int
  */
int main(void)
{

  /* USER CODE BEGIN 1 */

  /* USER CODE END 1 */

  /* MCU Configuration--------------------------------------------------------*/

  /* Reset of all peripherals, Initializes the Flash interface and the Systick. */
  HAL_Init();

  /* USER CODE BEGIN Init */

  /* USER CODE END Init */

  /* Configure the system clock */
  SystemClock_Config();

  /* USER CODE BEGIN SysInit */

  /* USER CODE END SysInit */

  /* Initialize all configured peripherals */
  MX_GPIO_Init();
  MX_DMA_Init();
  MX_CAN_Init();
  MX_ADC1_Init();
  /* USER CODE BEGIN 2 */
  	 CAN_transmit_init();
     HAL_CAN_Start(&hcan);

     HAL_ADC_Start_DMA(&hadc1, VR, 1);

  /* USER CODE END 2 */

  /* Infinite loop */
  /* USER CODE BEGIN WHILE */
  while (1)
  {
	  now = HAL_GetTick();

	 	       // ===== Xử lý nhấp nháy LED PB11 =====
	 	       if (blink_pb11 && (now - last_toggle_pb11 >= 500)) {
	 	           last_toggle_pb11 = now;
	 	           pb11_state ^= 1;
	 	           HAL_GPIO_WritePin(GPIOB, GPIO_PIN_11, pb11_state ? GPIO_PIN_SET : GPIO_PIN_RESET);
	 	       } else if (!blink_pb11) {
	 	           HAL_GPIO_WritePin(GPIOB, GPIO_PIN_11, GPIO_PIN_RESET);
	 	           pb11_state = 0;
	 	       }

	 	       // ===== Xử lý nhấp nháy LED PB10 =====
	 	       if (blink_pb10 && (now - last_toggle_pb10 >= 500)) {
	 	           last_toggle_pb10 = now;
	 	           pb10_state ^= 1;
	 	           HAL_GPIO_WritePin(GPIOB, GPIO_PIN_10, pb10_state ? GPIO_PIN_SET : GPIO_PIN_RESET);
	 	       } else if (!blink_pb10) {
	 	           HAL_GPIO_WritePin(GPIOB, GPIO_PIN_10, GPIO_PIN_RESET);
	 	           pb10_state = 0;
	 	       }

	 	       // ===== Gửi CAN nếu ADC thấp =====
	 	       if (can_send_adc_low && HAL_CAN_GetTxMailboxesFreeLevel(&hcan) > 0) {
	 	           HAL_CAN_AddTxMessage(&hcan, &tx_header, tx_buffer, &TxMailbox);
	 	           can_send_adc_low = 0;
	 	       }

	 	       // ===== Gửi CAN nếu ADC cao =====
	 	       if (can_send_adc_high && HAL_CAN_GetTxMailboxesFreeLevel(&hcan) > 0) {
	 	          HAL_CAN_AddTxMessage(&hcan, &tx_header, tx_buffer, &TxMailbox);
	 	           can_send_adc_high = 0;
	 	       }

	 	       // ===== Gửi CAN từ nút nhấn (PB5, PB6) nếu mailbox sẵn sàng =====
	 	       if ((blink_pb10 || blink_pb11) && HAL_CAN_GetTxMailboxesFreeLevel(&hcan) > 0) {
	 	    	  HAL_CAN_AddTxMessage(&hcan, &tx_header, tx_buffer, &TxMailbox);
	 	       }
    /* USER CODE END WHILE */

    /* USER CODE BEGIN 3 */
  }
  /* USER CODE END 3 */
}
